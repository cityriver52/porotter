1 : 0; const bStarts = needle && bName.startsWith(needle) ? 1 : 0; return bStarts - aStarts || Number(b.count || 0) - Number(a.count || 0) || String(a.name).localeCompare(String(b.name), 'ja'); }) .slice(0, 8); } function tagAutocompleteConfig(mode) { return mode === 'post' ? parseTagInput(input.value).filter(tag => tag.toLocaleLowerCase() !== query.toLocaleLowerCase()) : []; const matches = matchingTags(state.tags, query, excluded); state.tagSuggestionIndex[mode] = -1; list.innerHTML = matches.map(tag => ` `).join(''); list.hidden = matches.length === 0; input.setAttribute('aria-expanded', String(matches.length > 0)); } function closeTagSuggestions(mode) { const modes = mode ? [mode] : ['post', 'search']; modes.forEach(item => { const config = tagAutocompleteConfig(item); queryOne(config.list).hidden = true; queryOne(config.input).setAttribute('aria-expanded', 'false'); state.tagSuggestionIndex[item] = -1; }); } function chooseTagSuggestion(mode, tag) { const config = tagAutocompleteConfig(mode); const input = queryOne(config.input); input.value = replaceCurrentTag(input.value, tag, config.multiple); if (mode === 'post') syncQuickTagsFromInput(); closeTagSuggestions(mode); input.focus(); } function selectTagSuggestion(event, mode) { const button = event.target.closest('[data-tag-suggestion]'); if (button) chooseTagSuggestion(mode, button.dataset.tagSuggestion); } function handleTagSuggestionKeys(event, mode) { const config = tagAutocompleteConfig(mode); const list = queryOne(config.list); if (event.key === 'Escape') return closeTagSuggestions(mode); if (list.hidden && event.key === 'ArrowDown') renderTagSuggestions(mode); const options = queryAll('[data-tag-suggestion]', list); if (!options.length) return; if (event.key === 'ArrowDown' || event.key === 'ArrowUp') { event.preventDefault(); const direction = event.key === 'ArrowDown' ? 1 : -1; state.tagSuggestionIndex[mode] = (state.tagSuggestionIndex[mode] + direction + options.length) % options.length; options.forEach((option, index) => { const active = index === state.tagSuggestionIndex[mode]; option.classList.toggle('active', active); option.setAttribute('aria-selected', String(active)); }); options[state.tagSuggestionIndex[mode]].scrollIntoView({ block: 'nearest' }); } else if (event.key === 'Enter' && state.tagSuggestionIndex[mode] >= 0) { event.preventDefault(); chooseTagSuggestion(mode, options[state.tagSuggestionIndex[mode]].dataset.tagSuggestion); } }
